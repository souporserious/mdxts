# Tests

MDXTS ships with a test runner that can be used to run tests against your code and components. The test runner is built on top of [Jest](https://jestjs.io/), [Testing Library](https://testing-library.com/), and [Playwright](https://playwright.dev/).

```Button.tsx live
import { expect, test } from 'mdxts'

/** Used for taking actions and navigating. */
export function Button({
  label,
  onAction,
}: {
  /** Provides a label for the button. */
  label: string

  /** Called when an action for this button is requested. */
  onAction: () => void
}) {
  return <button onClick={onAction}>{label}</button>
}

test('should render', async ({ mount }) => {
  const component = await mount(<Button label="Hello Button" />)
  await expect(component).toContainText('Hello Button')
})
```
